\newcommand{\swline}[6]{
  \ifnum #6 > #4
    \newcount \disa
    \disa = #4
    \advance \disa by 10
    \ifnum \disa > #6
      \fontsize{#5}{#5}
      \selectfont

      \node at (0, -#4)
      [sloped, xslant=0]  (mid) { #2 };
      \node  
      [left = 0.0\the#5 cm of mid]
      [sloped, xslant=1] (lef) { #1 };
      \node 
      [sloped, xslant=-1] 
      [right = 0.0\the#5 cm of mid] (rig) { #3 };
    \fi
  \fi
  \advance #4 by 1;
  \advance #5 by 5;
}
